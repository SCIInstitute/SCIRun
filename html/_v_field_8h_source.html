107  {
108  if (basis_order_ == -1)
109  {
110  VMesh::dimension_type dim;
111  dim.resize(1); dim[0] = 0;
112  vfdata_->resize_fdata(dim);

117  VMesh::dimension_type dim;

122  {
123  VMesh::dimension_type dim;
