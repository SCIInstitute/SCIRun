392 template<class PTYPE> inline bool Bundle::NrrdToMatrixHelper(
393  NrrdDataHandle dataH, MatrixHandle& matH)
394 {
395  if (dataH->nrrd_->dim == 1)
396  {
397  int cols = dataH->nrrd_->axis[0].size;
398 
399  ColumnMatrix* matrix = new ColumnMatrix(cols);
400 
401  PTYPE *val = reinterpret_cast<PTYPE*>(dataH->nrrd_->data);
402  double *data = matrix->get_data_pointer();
403 
404  for(int c=0; c<cols; c++)
405  {
406  *data = *val;
407  data++;
408  val++;
409  }
410  matH = matrix;
411  return(true);
412  }
413 
414  if (dataH->nrrd_->dim == 2)
415  {
416  if (transposeNrrd_)
417  {
418  int rows = dataH->nrrd_->axis[1].size;
419  int cols = dataH->nrrd_->axis[0].size;
420 
421  DenseMatrix* matrix = new DenseMatrix(rows,cols);
422 
423  PTYPE *val = reinterpret_cast<PTYPE*>(dataH->nrrd_->data);
424  double *data = matrix->get_data_pointer();
425 
426  int i,j;
427  i = 0; j = 0;
428  for(int r=0; r<rows; r++)
429  {
430  for(int c=0; c<cols; c++)
431  {
432  i = c + cols*r;
433  data[j++] = val[i];
434  }
435  }
436  matH = matrix;
437  }
438  else
439  {
440  int cols = dataH->nrrd_->axis[1].size;
441  int rows = dataH->nrrd_->axis[0].size;
442 
443  DenseMatrix* matrix = new DenseMatrix(cols,rows);
444 
445  PTYPE *val = reinterpret_cast<PTYPE*>(dataH->nrrd_->data);
446  double *data = matrix->get_data_pointer();
447 
448  for(int c=0; c<cols; c++)
449  {
450  for(int r=0; r<rows; r++)
451  {
452  *data++ = *val++;
453  }
454  }
455  matH = matrix;
456  }
457  return(true);
458  }
459  // Improper dimensions
460  return(false);
461 }
