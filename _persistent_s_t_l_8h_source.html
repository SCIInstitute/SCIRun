72 // Persistent io for maps
73 template <class Key, class Data>
74 inline void
75 Pio(Piostream& stream, std::map<Key, Data>& data) {
76 
77  stream.begin_class("Map", MAP_VERSION);
78 
79  // if reading from stream
80  if (stream.reading()) {
81  // get map size
82  int n;
83  Pio(stream, n);
84  // read elements
85  Key k;
86  Data d;
87  for (int i = 0; i < n; i++) {
88  Pio(stream, k);
89  Pio(stream, d);
90  data[k] = d;
91  }
92 
93  }
94  // if writing to stream
95  else {
96  // write map size
97  int n = data.size();
98  Pio(stream, n);
99  // write elements
100  for (auto iter = data.begin(); iter != data.end(); iter++) {
101  // have to copy iterator elements,
102  // since passing them directly in a
103  // call to Pio can be invalid because
104  // Pio passes data by reference
105  Key ik = (*iter).first;
106  Data dk = (*iter).second;
107  Pio(stream, ik);
108  Pio(stream, dk);
109  }
110 
111  }
112 
113  stream.end_class();
114 
115 }

147 template <class T>
148 void Pio(Piostream& stream, std::vector<T>& data)
149 {
150  if (stream.reading() && stream.peek_class() == "Array1")
151  {
152  stream.begin_class("Array1", STLVECTOR_VERSION);
153  }
154  else
155  {
156  stream.begin_class("STLVector", STLVECTOR_VERSION);
157  }
158 
159  int size=static_cast<int>(data.size());
160  stream.io(size);
161 
162  if(stream.reading()){
163  data.resize(size);
164  }
165 
166  for (int i = 0; i < size; i++)
167  {
168  Pio(stream, data[i]);
169  }
170 
171  stream.end_class();
172 }

208 void Pio(Piostream& stream, std::list<T>& data)
209 {
210  stream.begin_cheap_delim();

212  int size=data.size();
213  stream.io(size);

215  if(stream.reading()){
216  data.resize(size);
217  }
218 
219  for (typename std::list<T>::iterator ii=data.begin(); ii!=data.end(); ii++)
220  Pio(stream, *ii);
