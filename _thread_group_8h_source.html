63 DESCRIPTION
64  A group of threads that are linked together for scheduling
65  and control purposes. The threads may be stopped, resumed
66  and alerted simultaneously.
67 
68 ****************************************/
69  class SCISHARE ThreadGroup {
70  public:
71 
73  // Create a thread group with the specified <i>name</i>.
74  // <i>parentGroup</i> specifies the parent <b>ThreadGroup</b>
75  // which defaults to the default top-level group.
76  ThreadGroup(const char* name, ThreadGroup* parentGroup=0);
77 
79  // Destroy the thread group. All of the running threads
80  // should be stopped before the <b>ThreadGroup</b> is destroyed.
81  ~ThreadGroup();
82 
84  // Return a snapshot of the number of living threads. If
85  // <i>countDaemon</i> is true, then daemon threads will be
86  // included in the count.
87  int numActive(bool countDaemon);
88 
90  // Stop all of the threads in this thread group
91  void stop();
92 
94  // Resume all of the threads in this thread group
95  void resume();
96 
98  // Wait until all of the threads have completed.
99  void join();
100 
102  // Detach the thread, joins are no longer possible.
103  void detach();
104 
106  // Return the parent <b>ThreadGroup.</b> Returns null if
107  // this is the default threadgroup.
108  ThreadGroup* parentGroup();
109 
111  // Arrange to have the threadGroup gang scheduled, so that
112  // all of the threads will be executing at the same time if
113  // multiprocessing resources permit. This interface will
114  // typically be employed by the <i>Thread::parallel</i>
115  // static method, and will typically not be called directly
116  // by user code. Threads added to the group after this call
117  // may or may not be included in the schedule gang.
118  void gangSchedule();
119 
120  protected:
121  friend class Thread;
122  static ThreadGroup* s_default_group;
123 
124  private:
125  Mutex lock_;
126  const char* name_;
127  ThreadGroup* parent_;
128  std::vector<ThreadGroup*> groups_;
129  std::vector<Thread*> threads_;
130  void addme(ThreadGroup* t);
131  void addme(Thread* t);
132 
133  // Cannot copy them
134  ThreadGroup(const ThreadGroup&);
135  ThreadGroup& operator=(const ThreadGroup&);
136  };
