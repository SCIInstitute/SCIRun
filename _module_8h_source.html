293  template <class T, size_t N>
294  std::vector<boost::shared_ptr<T>> Module::getRequiredDynamicInputs(const DynamicPortName<T,N>& port)
295  {
296  auto handleOptions = get_dynamic_input_handles(port.id_);
297  std::vector<boost::shared_ptr<T>> handles;
298  auto check = [&, this](SCIRun::Core::Datatypes::DatatypeHandleOption opt) { return this->checkInput<T>(opt, port.id_); };
299  auto end = handleOptions.end() - 1; //leave off empty final port
300  std::transform(handleOptions.begin(), end, std::back_inserter(handles), check);
301  return handles;
302  }

324  template <class T>
325  boost::shared_ptr<T> Module::checkInput(SCIRun::Core::Datatypes::DatatypeHandleOption inputOpt, const PortId& id)
326  {
327  if (!*inputOpt)
328  MODULE_ERROR_WITH_TYPE(NullHandleOnPortException, "Null handle on port " + id.name);
329 
330  boost::shared_ptr<T> data = boost::dynamic_pointer_cast<T>(*inputOpt);
331  if (!data)
332  {
333  std::ostringstream ostr;
334  ostr << "Wrong datatype on port #" << id.name << "; expected " << typeid(T).name() << " but received " << typeid(*inputOpt).name();
335  MODULE_ERROR_WITH_TYPE(WrongDatatypeOnPortException, ostr.str());
336  }
337  return data;
338  }
