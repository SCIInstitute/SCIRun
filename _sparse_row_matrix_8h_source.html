56  //Legacy construction compatibility. Useful for converting old code, but should be avoided in new code.
57  SparseRowMatrixGeneric(int nrows, int ncols, const index_type* rowCounter, const index_type* columnCounter, size_t nnz) : EigenBase(nrows, ncols)
58  {
59  if (rowCounter[nrows] != nnz)
60  THROW_INVALID_ARGUMENT("Invalid sparse row matrix array: row accumulator array does not match number of non-zero elements.");
61  std::vector<Triplet> triplets;
62  triplets.reserve(nnz);
63 
64  int i = 0;
65  int j = 0;
66  while (i < nrows)
67  {
68  while (j < rowCounter[i + 1])
69  {
70  index_type column = columnCounter[j];
71  if (column >= ncols)
72  THROW_INVALID_ARGUMENT("Invalid sparse row matrix array: column index out of bounds.");
73  triplets.push_back(Triplet(i, columnCounter[j], 0));
74  j++;
75  }
76  i++;
77  }
78  this->setFromTriplets(triplets.begin(), triplets.end());
79  }

294 template <typename T> inline bool isPositiveDefiniteMatrix (const SCIRun::Core::Datatypes::SparseRowMatrixGeneric<T> & m)
295 {
296  if (!isSymmetricMatrix(m)) return false; //a matrix must be symmetric to be positive definite
297 
298  if ( !ContainsValidValues(m) ) return false;
299 
300  for (int k = 0; k < m.outerSize(); ++k) //all diagonal elements are positive?
301  if ( m.coeff(k, k) <= 0 ) return false;
302 
303  for (int k = 0; k < m.outerSize(); ++k)
304  {
305  double tmp1=0.0, tmp2=0.0;
306  for (typename SCIRun::Core::Datatypes::SparseRowMatrixGeneric<T>::InnerIterator it(m,k); it; ++it)
307  {
308  if (it.col()!=it.row())
309  {
310  tmp1+=std::fabs(m.coeff(it.row(),it.col())); //abs. sum over col
311  tmp2+=std::fabs(m.coeff(it.col(),it.row())); //abs. sum over row
312  }
313  }
314 
315  if ( !((tmp1<m.coeff(k, k)) && (tmp2<m.coeff(k, k))) ) return false;
316  }
317 
318  return true;
319 }
