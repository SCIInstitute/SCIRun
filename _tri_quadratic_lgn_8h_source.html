91  w[1] * cd.node1() +
92  w[2] * cd.node2() +

109  w[0]*cd.node0()

111  + w[2]*cd.node2()
112  + w[3]*this->nodes_[cd.edge0_index()]
113  + w[4]*this->nodes_[cd.edge1_index()]
