88  T interpolate(const VECTOR &coords, const ElemData &cd) const
89  {
90  double w[10];

92 
93  return (T)(w[0] * cd.node0() +
94  w[1] * cd.node1() +
95  w[2] * cd.node2() +
96  w[3] * cd.node3() +
97  w[4] * this->nodes_[cd.edge0_index()] +
98  w[5] * this->nodes_[cd.edge1_index()] +
99  w[6] * this->nodes_[cd.edge2_index()] +
