117 template <class HType, class PortTag>
118 void
119 GenericReader<HType, PortTag>::execute()
120 {
121 #ifdef SCIRUN4_CODE_TO_BE_ENABLED_LATER
122  bool filename_changed = gui_filename_.changed();
123 
124  if (gui_from_env_.get() != "")
125  {
126  std::string filename_from_env = gui_from_env_.get();
127  if (sci_getenv(filename_from_env))
128  {
129  std::string envfilename = sci_getenv(filename_from_env);
130  gui_filename_.set(envfilename);
131  get_ctx()->reset();
132  filename_changed = true;
133  }
134  }
135 #endif
136 
137  // If there is an optional input string set the filename to it in the GUI.
138  //TODO: this will be a common pattern for file loading. Perhaps it will be a base class method someday...
139  auto fileOption = getOptionalInput(Filename);
140  if (!fileOption)
141  filename_ = get_state()->getValue(SCIRun::Core::Algorithms::Variables::Filename).getString();
142  else
143  filename_ = (*fileOption)->value();
144 
145  // Read the status of this file so we can compare modification timestamps
146 
147  if(filename_.empty())
148  {
149  error("No file has been selected. Please choose a file.");
150  return;
151  }
152  else if (!file_exists(filename_))
153  {
154  if (!importing_)
155  {
156  error("File '" + filename_ + "' not found.");
157  return;
158  }
159  else
160  {
161  warning("File '" + filename_ + "' not found. Maybe the plugin can find it?");
162  }
163  }
164 
165  // If we haven't read yet, or if it's a new filename,
166  // or if the datestamp has changed -- then read...
167 
168  time_t new_filemodification = boost::filesystem::last_write_time(filename_);
169 
170  if(
171 #ifdef SCIRUN4_CODE_TO_BE_ENABLED_LATER
172  inputs_changed_ || filename_changed ||
173  new_filemodification != old_filemodification_ ||
174  !oport_cached(0) ||
175  !oport_cached("Filename")
176 #else
177  true
178 #endif
179  )
180  {
181 #ifdef SCIRUN4_CODE_TO_BE_ENABLED_LATER
182  update_state(Executing);
183 #endif
184  old_filemodification_ = new_filemodification;
185 
186  HType handle;
187 
188  remark("loading file " +filename_);
189 
190  if (importing_)
191  {
192  if (!call_importer(filename_, handle))
193  {
194  error("Import failed.");
195  return;
196  }
197  }
198  else
199  {
200  PiostreamPtr stream = auto_istream(filename_, getLogger());
201  if (!stream)
202  {
203  error("Error reading file '" + filename_ + "'.");
204  return;
205  }
206 
207  // Read the file
208  Pio(*stream, handle);
209 
210  if (!handle || stream->error())
211  {
212  error("Error reading data from file '" + filename_ +"'.");
213  return;
214  }
215  }
216 
217  Core::Datatypes::StringHandle shandle(new Core::Datatypes::String(filename_));
218  sendOutput(FileLoaded, shandle);
219  sendOutput(objectPortName_, handle);
220  }
221 
222 }
