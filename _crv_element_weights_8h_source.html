77  template <class VECTOR>
78  void get_cubic_weights(const VECTOR &coords, double *w) const
79  {
80  const double x = static_cast<double>(coords[0]);
81  w[0] = (x-1)*(x-1)*(1 + 2*x);
82  w[1] = (x-1)*(x-1)*x;
83  w[2] = (3 - 2*x)*x*x;
84  w[3] = (-1+x)*x*x;
