48  {
49  public:
50  explicit ModuleConsumer(ModuleWorkQueuePtr workQueue, const Networks::ExecutableLookup* lookup, ProducerInterfacePtr producer) :
51  work_(workQueue), producer_(producer), lookup_(lookup), shouldLog_(false)
52  {
53  log_.setVerbose(shouldLog_);
54  if (shouldLog_)
55  log_ << Core::Logging::DEBUG_LOG << "Consumer created." << std::endl;
56  }
57  void operator()() const
58  {
59  if (!producer_)
60  {
61  if (shouldLog_)
62  log_ << Core::Logging::DEBUG_LOG << "Consumer quitting due to no producer pointer." << std::endl;
63  return;
64  }
65 
66  log_ << Core::Logging::DEBUG_LOG << "Consumer started." << std::endl;
67 
68  while (!producer_->isDone() || moreWork())
69  {
70  {
71  if (shouldLog_)
72  log_ << Core::Logging::DEBUG_LOG << "\tConsumer thinks work queue is not empty.";
73 
74  if (shouldLog_)
75  log_ << Core::Logging::DEBUG_LOG << "\tConsumer accessing front of work queue.";
76  Networks::ModuleHandle unit;
77  work_->pop(unit);
78  if (shouldLog_)
79  log_ << Core::Logging::DEBUG_LOG << "\tConsumer popping front of work queue.";
80 
81  if (unit)
82  {
83  if (shouldLog_)
84  log_ << Core::Logging::DEBUG_LOG << "~~~Processing " << unit->get_id();
85 
86  ModuleExecutor executor(unit, lookup_, producer_);
87  //TODO: thread pool
88  boost::thread t(boost::bind(&ModuleExecutor::run, executor));
89  }
90  else
91  {
92  if (shouldLog_)
93  log_ << Core::Logging::DEBUG_LOG << "\tConsumer received null module";
94  }
95  }
96  }
97  log_ << Core::Logging::DEBUG_LOG << "Consumer done." << std::endl;
98  }
99 
100  bool moreWork() const
101  {
102  return !work_->empty();
103  }
