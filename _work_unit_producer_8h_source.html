57  virtual void enqueueReadyModules() const
58  {
59  if (!isDone())
60  {
61  ParallelModuleExecutionOrder order;
62  {
63  Core::Thread::Guard g(lock_->get());
64  order = scheduler_.schedule(*network_);
65  }
66  log_ << Core::Logging::DEBUG_LOG << "Producer processing min group " << order.minGroup();
67  auto groupIter = order.getGroup(order.minGroup());
68  BOOST_FOREACH(const ParallelModuleExecutionOrder::ModulesByGroup::value_type& mod, groupIter)
69  {
70  auto module = network_->lookupModule(mod.second);
71 
72  if (module->executionState() != Networks::ModuleInterface::Executing)
73  {
74  log_ << Core::Logging::DEBUG_LOG << "Producer pushing module " << mod.second << std::endl;
75  work_->push(module);
76  doneCount_.fetch_add(1);
77  log_ << Core::Logging::DEBUG_LOG << "Producer status: " << doneCount_ << " out of " << network_->nmodules() << std::endl;
78  }
79  }
80  }
81  }
82 
83  void operator()() const
84  {
85  ScopedExecutionBoundsSignaller signaller(bounds_, [=]() { return lookup_->errorCode(); });
86 
87  log_ << Core::Logging::DEBUG_LOG << "Producer started" << std::endl;
88 
89  enqueueReadyModules();
90 
91  while (!isDone())
92  boost::this_thread::sleep(boost::posix_time::milliseconds(1000));
93 
94  log_ << Core::Logging::DEBUG_LOG << "Producer is done." << std::endl;
95  }
96 
97  bool isDone() const
98  {
99  return doneCount_ >= network_->nmodules();
100  }
