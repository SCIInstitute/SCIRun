60  virtual void enqueueReadyModules() const
61  {
62  Core::Thread::Guard g(lock_->get());
63  if (!isDone())
64  {
65  auto order = scheduler_.schedule(*network_);
66  if (shouldLog_)
67  {
68  std::ostringstream ostr;
69  ostr << "Producer received this ordering: \n" << order << std::endl;
70  log_ << Core::Logging::DEBUG_LOG << ostr.str() << std::endl;
71  log_ << Core::Logging::DEBUG_LOG << "Producer processing min group " << order.minGroup();
72  }
73  auto groupIter = order.getGroup(order.minGroup());
74  BOOST_FOREACH(const ParallelModuleExecutionOrder::ModulesByGroup::value_type& mod, groupIter)
75  {
76  auto module = network_->lookupModule(mod.second);
77 
78  if (module->executionState() == Networks::ModuleInterface::Waiting)
79  {
80  if (shouldLog_)
81  log_ << Core::Logging::DEBUG_LOG << "Producer pushing module " << mod.second << std::endl;
82 
83  if (doneIds_.find(mod.second) != doneIds_.end())
84  {
85  if (shouldLog_)
86  SCIRun::Core::Logging::Log::get() << SCIRun::Core::Logging::INFO << "Module producer: wants to enqueue module " << mod.second << " a second time." << std::endl;
87  }
88  else
89  {
90  work_->push(module);
91  doneIds_.insert(mod.second);
92  doneCount_.fetch_add(1);
93 
94  if (shouldLog_)
95  log_ << Core::Logging::DEBUG_LOG << "Producer status: " << doneCount_ << " out of " << numModules_ << std::endl;
96  }
97  }
98  }
99  }
100  }
