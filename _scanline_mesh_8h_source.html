439  template<class VECTOR>
440  bool get_coords(VECTOR &coords, const Core::Geometry::Point &p, typename Elem::index_type idx) const
441  {
442  // If polynomial order is larger, use the complicated HO basis implementation
443  // Since this is a latvol and most probably linear, this function is to expensive
444  if (basis_.polynomial_order() > 1)
445  {
446  ElemData ed(*this, idx);
447  return (basis_.get_coords(coords, p, ed));
448  }
449 
450  // Cheap implementation that assumes it is a regular grid
451  // This implementation should be faster then the interpolate of the linear
452  // basis which needs to work as well for the unstructured hexvol :(
453  const Core::Geometry::Point r = transform_.unproject(p);
454 
455  coords.resize(1);
456  coords[0] = static_cast<typename VECTOR::value_type>(r.x()-static_cast<double>(idx));
457 
458  if (static_cast<double>(coords[0]) < 0.0) if (static_cast<double>(coords[0]) > -(1e-8))
459  coords[0] = static_cast<typename VECTOR::value_type>(0.0); else return (false);
460  if (static_cast<double>(coords[0]) > 1.0) if (static_cast<double>(coords[0]) < 1.0+(1e-8))
461  coords[0] = static_cast<typename VECTOR::value_type>(1.0); else return (false);
462 
463  return (true);
464  }

521  template<class VECTOR>
522  void jacobian(const VECTOR& coords, typename Elem::index_type idx, double* J) const
523  {
524  if (basis_.polynomial_order() > 1)
525  {
526  StackVector<Core::Geometry::Point,3> Jv;
527  ElemData ed(*this,idx);
528  basis_.derivate(coords,ed,Jv);
529  Core::Geometry::Vector Jv1, Jv2;
530  Core::Geometry::Vector(Jv[0]).find_orthogonal(Jv1,Jv2);
531  J[0] = Jv[0].x();
532  J[1] = Jv[0].y();
533  J[2] = Jv[0].z();
534  J[3] = Jv1.x();
535  J[4] = Jv1.y();
536  J[5] = Jv1.z();
537  J[6] = Jv2.x();
538  J[7] = Jv2.y();
539  J[8] = Jv2.z();
540 
541  return;
542  }

649  template <class INDEX>
650  bool
651  find_closest_node(double& pdist, Core::Geometry::Point &result,
652  INDEX &node, const Core::Geometry::Point &p) const
653  {
655  if (ni_ == 0) return (false);
656 
657  const Core::Geometry::Point r = transform_.unproject(p);
658 
659  double rx = floor(r.x() + 0.5);
660  const double nii = static_cast<double>(ni_-1);
661 
662  if (rx < 0.0) rx = 0.0; if (rx > nii) rx = nii;
663 
664  result = transform_.project(Core::Geometry::Point(rx,0.0,0.0));
665  index_type irx =static_cast<index_type>(rx);
666  node = INDEX(irx);
667  pdist = (p-result).length();
668 
669  return (true);
670  }
