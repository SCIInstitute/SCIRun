83  template <typename T>
84  std::istream& operator>>(std::istream& istr, DenseMatrixGeneric<T>& m)
85  {
86  std::vector<std::vector<T> > values;
87 
88  std::string line;
89 
90  while (!std::getline(istr, line, '\n').eof())
91  {
92  std::istringstream reader(line);
93 
94  std::vector<T> lineData;
95 
96  while (!reader.eof())
97  {
98  T val;
99  reader >> val;
100 
101  if (reader.fail())
102  {
103  FloatNaNHelper<T> helper(val);
104  reader >> helper;
105 
106  if (reader.fail())
107  {
108  //THROW_INVALID_ARGUMENT("Matrix reading failed: stream failed");
109  break;
110  }
111  }
112 
113  lineData.push_back(val);
114  }
115 
116  if (!lineData.empty())
117  {
118  if (!values.empty() && values.back().size() != lineData.size())
119  THROW_INVALID_ARGUMENT("Improper format of matrix text stream: not every line contains the same amount of numbers.");
120 
121  values.push_back(lineData);
122  }
123  }
124 
125  m.resize(values.size(), values[0].size());
126  m.setZero();
127  for (int i = 0; i < m.rows(); ++i)
128  {
129  for (int j = 0; j < m.cols(); ++j)
130  {
131  m(i,j) = values[i][j];
132  }
133  }
134 
135  return istr;
136  }

148  template <typename T>
149  void DenseMatrixGeneric<T>::io(Piostream& stream)
150  {
151  int version=stream.begin_class("DenseMatrix", DENSEMATRIX_VERSION);
152  // Do the base class first...
153  MatrixIOBase::io(stream);
154 
155  if (version < 4)
156  {
157  int nrows = static_cast<int>(this->rows());
158  int ncols = static_cast<int>(this->cols());
159  stream.io(nrows);
160  stream.io(ncols);
161  this->resize(nrows, ncols);
162  }
163  else
164  {
165  long long nrows = static_cast<long long>(this->rows());
166  long long ncols = static_cast<long long>(this->cols());
167  stream.io(nrows);
168  stream.io(ncols);
169  if(stream.reading())
170  {
171  this->resize(nrows, ncols);
172  }
173  }
174 
175  stream.begin_cheap_delim();
176 
177  int split;
178  if (stream.reading())
179  {
180  if (version > 2)
181  {
182  split = this->separate_raw_;
183  Pio(stream, split);
184  if (this->separate_raw_)
185  {
186  Pio(stream, this->raw_filename_);
187  FILE *f=fopen(this->raw_filename_.c_str(), "r");
188  if (f)
189  {
190  fread(this->data(), sizeof(T), this->rows() * this->cols(), f);
191  fclose(f);
192  }
193  else
194  {
195  const std::string errmsg = "Error reading separated file '" + this->raw_filename_ + "'";
196  std::cerr << errmsg << "\n";
197  BOOST_THROW_EXCEPTION(SCIRun::Core::ExceptionBase() << FileNotFound(errmsg));
198  }
199  }
200  }
201  else
202  {
203  this->separate_raw_ = false;
204  }
205  split = this->separate_raw_;
206  }
207  else
208  { // writing
209  std::string filename = this->raw_filename_;
210  split = this->separate_raw_;
211  if (split)
212  {
213  if (filename == "")
214  {
215  if (stream.file_name.c_str())
216  {
217  size_t pos = stream.file_name.rfind('.');
218  if (pos == std::string::npos) filename = stream.file_name + ".raw";
219  else filename = stream.file_name.substr(0,pos) + ".raw";
220  }
221  else
222  {
223  split=0;
224  }
225  }
226  }
227  Pio(stream, split);
228  if (split)
229  {
230  Pio(stream, filename);
231  FILE *f = fopen(filename.c_str(), "w");
232  fwrite(this->data(), sizeof(T), this->rows() * this->cols(), f);
233  fclose(f);
234  }
235  }
236 
237  if (!split)
238  {
239  size_t block_size = this->rows() * this->cols();
240  if (!stream.block_io(this->data(), sizeof(T), block_size))
241  {
242  for (size_t i = 0; i < block_size; i++)
243  {
244  stream.io(this->data()[i]);
245  }
246  }
247  }
248  stream.end_cheap_delim();
249  stream.end_class();
250  }
